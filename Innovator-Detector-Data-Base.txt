-- Crear la base de datos Innovator-Detector
CREATE DATABASE IF NOT EXISTS Innovator_Detector;

-- Usar la base de datos Innovator-Detector
USE Innovator_Detector;

-- Crear la tabla 'user'
CREATE TABLE user (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    rol INT NOT NULL,
    gestor INT,
    FOREIGN KEY (rol) REFERENCES rol(id),
    FOREIGN KEY (gestor) REFERENCES gestor(id)
);

-- Crear la tabla 'gestor'
CREATE TABLE gestor (
    id INT AUTO_INCREMENT PRIMARY KEY
);

-- Crear la tabla 'enterprise'
CREATE TABLE enterprise (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    address VARCHAR(255),
    contact VARCHAR(255)
);

-- Crear la tabla 'rol'
CREATE TABLE rol (
    id INT AUTO_INCREMENT PRIMARY KEY,
    rol_name VARCHAR(255) NOT NULL,
    description TEXT
);

-- Crear la tabla 'domain'
CREATE TABLE domain (
    id INT AUTO_INCREMENT PRIMARY KEY,
    domain_name VARCHAR(255) NOT NULL,
    description TEXT
);

-- Crear la tabla 'pregunta'
CREATE TABLE pregunta (
    id INT AUTO_INCREMENT PRIMARY KEY,
    pregunta TEXT NOT NULL
);

-- Crear la tabla 'valor'
CREATE TABLE valor (
    id INT AUTO_INCREMENT PRIMARY KEY,
    valor DECIMAL(10, 2) -- Cambiar el tipo de dato seg√∫n tus necesidades
);

-- Crear la tabla 'answer'
CREATE TABLE answer (
    date DATE NOT NULL,
    description TEXT
);
